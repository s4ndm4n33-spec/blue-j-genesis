5. Backend: api-server/src/routes/bluej/prettier-format.ts (NEW FILE)
import { Router, type IRouter } from "express";
import { execFile } from "child_process";
import { promisify } from "util";
import { writeFile, unlink } from "fs/promises";
import { join } from "path";
import { randomUUID } from "crypto";
const execFileAsync = promisify(execFile);
const router: IRouter = Router();
const TMP_DIR = "/tmp/bluej-prettier";
function resolveParser(language: string): string | null {
  const l = language.toLowerCase();
  if (l === "javascript" || l === "js") return "babel";
  if (l === "typescript" || l === "ts") return "typescript";
  if (l === "json") return "json";
  if (l === "css" || l === "scss") return "css";
  if (l === "html" || l === "htm") return "html";
  if (l === "markdown" || l === "md") return "markdown";
  if (l === "python" || l === "py") return null;
  return "babel";
}
router.post("/", async (req, res) => {
  try {
    const { code, language = "javascript" } = req.body;
    if (!code?.trim()) { res.status(400).json({ error: "No code provided" }); return; }
    const parser = resolveParser(language);
    if (!parser) {
      res.json({ formatted: code, error: `Prettier does not support ${language}.`, changed: false, parser: null });
      return;
    }
    const id = randomUUID();
    const ext = parser === "typescript" ? "ts" : parser === "json" ? "json" : "js";
    const file = join(TMP_DIR, `${id}.${ext}`);
    await writeFile(file, code, "utf-8");
    try {
      const { stdout } = await execFileAsync(
        "npx",
        ["prettier", "--parser", parser, "--print-width", "100", "--tab-width", "2", "--single-quote", "--semi", file],
        { timeout: 5000, cwd: "/tmp" }
      );
      await unlink(file).catch(() => {});
      const changed = stdout !== code;
      res.json({ formatted: stdout, changed, parser, error: null });
    } catch (err: any) {
      await unlink(file).catch(() => {});
      const stderr = err?.stderr || err?.message || "";
      res.json({ formatted: code, changed: false, parser, error: `Prettier failed: ${stderr.slice(0, 300)}` });
    }
  } catch (err) {
    req.log.error({ err }, "Prettier format error");
    res.status(500).json({ error: "Format failed" });
  }
});
export default router;

const handleFormat = async () => {
  if (!myCode.trim()) return;
  setFormatting(true);
  try {
    const resp = await fetch('/api/bluej/prettier', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ code: myCode, language: selectedLanguage }),
    });
    const data = await resp.json();
    if (data.formatted && data.changed) {
      setMyCode(data.formatted);
      addSystemMessage?.(`Prettier formatted your ${selectedLanguage} code.`);
    } else if (data.error) {
      addSystemMessage?.(`Prettier: ${data.error}`);
    } else {
      addSystemMessage?.(`Prettier: Code already formatted.`);
    }
  } catch (err) {
    addSystemMessage?.('Prettier formatting failed. Check connection.');
  } finally {
    setFormatting(false);
  }
};

// Read-only repo context for J (auto-fetches recent files)
router.get("/context/:sessionId", async (req, res) => {
  try {
    const rows = await db.select().from(gitRepos).where(eq(gitRepos.sessionId, req.params.sessionId));
    if (rows.length === 0) { res.json({ context: "" }); return; }
    const repo = rows[0];
    const cwd = repo.localPath;
    // Get recent commit files (last 10)
    const { stdout: log } = await execFileAsync("git", ["-C", cwd, "log", "--name-only", "--pretty=format:", "-10"]);
    const files = [...new Set(log.split("\n").filter((f) => f.trim() && !f.includes("Binary")))];
    const fileContents: string[] = [];
    let totalChars = 0;
    const MAX_CHARS = 4000;
    for (const f of files.slice(0, 15)) {
      const p = join(cwd, f);
      if (!p.startsWith(cwd)) continue;
      try {
        const c = await readFile(p, "utf-8");
        if (c.length > 500) continue;
        if (totalChars + c.length > MAX_CHARS) break;
        fileContents.push(`\n--- FILE: ${f} ---\n${c}`);
        totalChars += c.length + f.length + 20;
      } catch { /* skip unreadable */ }
    }
    const context = `Repository: ${repo.name}\nBranch: ${repo.branch}\n${fileContents.join("\n")}`;
    res.json({ context });
  } catch (err) {
    req.log.error({ err }, "Git context error");
    res.json({ context: "" });
  }
});

T004: Leaderboard / Gamification
1. Create src/lib/leaderboard-store.ts (NEW FILE)
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
export interface LeaderboardEntry {
  id: string;
  name: string;
  score: number;
  level: number;
  achievements: number;
  streak: number;
  updatedAt: number;
}
interface LeaderboardState {
  entries: LeaderboardEntry[];
  userId: string;
  userName: string;
  addOrUpdateUser: (score: number, level: number, achievements: number, streak: number) => void;
  setUserName: (name: string) => void;
  getRank: (id: string) => number;
}
export const useLeaderboardStore = create<LeaderboardState>()(
  persist(
    (set, get) => ({
      entries: [
        { id: 'demo-1', name: 'CodeWolf', score: 8500, level: 18, achievements: 9, streak: 12, updatedAt: Date.now() },
        { id: 'demo-2', name: 'NeonDev', score: 6200, level: 14, achievements: 7, streak: 5, updatedAt: Date.now() },
        { id: 'demo-3', name: 'SyntaxQueen', score: 4100, level: 10, achievements: 5, streak: 3, updatedAt: Date.now() },
        { id: 'demo-4', name: 'BugHunter', score: 2300, level: 6, achievements: 3, streak: 2, updatedAt: Date.now() },
        { id: 'demo-5', name: 'HelloWorld', score: 800, level: 2, achievements: 1, streak: 1, updatedAt: Date.now() },
      ],
      userId: 'me',
      userName: 'You',
      addOrUpdateUser: (score, level, achievements, streak) => {
        const { entries, userId, userName } = get();
        const filtered = entries.filter((e) => e.id !== userId);
        const updated: LeaderboardEntry = {
          id: userId, name: userName, score, level, achievements, streak, updatedAt: Date.now(),
        };
        const next = [...filtered, updated].sort((a, b) => b.score - a.score);
        set({ entries: next });
      },
      setUserName: (name) => set({ userName: name.trim() || 'You' }),
      getRank: (id) => {
        const idx = get().entries.findIndex((e) => e.id === id);
        return idx >= 0 ? idx + 1 : get().entries.length + 1;
      },
    }),
    { name: 'bluej-leaderboard' }
  )
);
